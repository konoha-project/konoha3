cmake_minimum_required(VERSION 2.6)
find_library(HAVE_LIBEVENT NAMES event)
find_library(HAVE_LIBJANSSON NAMES jansson)
if(HAVE_LIBEVENT)
	if(HAVE_LIBJANSSON)
		set(ExtraLibs ${ExtraLibs} ${HAVE_LIBEVENT})
		set(ExtraLibs ${ExtraLibs} ${HAVE_LIBJANSSON})
		set(PACKAGE_EXTRA_LIBRARY ${ExtraLibs})
		set(PACKAGE_SOURCE_CODE eventlistener_glue.c)
		set(PACKAGE_SCRIPT_CODE eventlistener_glue.k)
		add_konoha_package(dscript.eventlistener)
	endif(HAVE_LIBJANSSON)
endif(HAVE_LIBEVENT)
